71      private static final Logger log = LoggerFactory.getLogger(DefaultDocumentCommentingFieldPlugin.class);
72      private static final ResourceReference ADD_ICON_REF = new PackageResourceReference(
73              DefaultDocumentCommentingFieldPlugin.class, "add-small-16.png");
74      private static final ResourceReference EDIT_ICON_REF = new PackageResourceReference(
75              DefaultDocumentCommentingFieldPlugin.class, "edit-small-16.png");
76      private static final ResourceReference DELETE_ICON_REF = new PackageResourceReference(
77              DefaultDocumentCommentingFieldPlugin.class, "delete-small-16.png");

79      private final CommentingContext commentingContext;
80      private final long queryLimit;
81      private final List<CommentItem> currentCommentItems = new LinkedList<>();
82      private final DialogAction addDialogAction;
83      private final boolean editableByAuthorOnly;
84      private final boolean deletableByAuthorOnly;
85      private CommentPersistenceManager commentPersistenceManager;
86  
87      public DefaultDocumentCommentingFieldPlugin(IPluginContext context, IPluginConfig config) {
88          super(context, config);
89          setOutputMarkupId(true);
90          JcrNodeModel documentModel = (JcrNodeModel) getModel();
91          commentingContext = new CommentingContext(context, config, documentModel);
92          String commentPersistenceManagerClazz = config.getString("comment.persistence.manager", null);
93          if (StringUtils.isNotBlank(commentPersistenceManagerClazz)) {
94              try {
95                  commentPersistenceManager = (CommentPersistenceManager) Class.forName(commentPersistenceManagerClazz)
96                          .getDeclaredConstructor().newInstance();
97              } catch (Exception e) {
98                  log.error("Cannot create custom comment persistence manager.", e);
99              }
100         }
101         if (commentPersistenceManager == null) {
102             commentPersistenceManager = new DefaultJcrCommentPersistenceManager();
103         }
104         queryLimit = config.getAsLong("comment.query.limit", 100);
105         editableByAuthorOnly = config.getAsBoolean("comment.editable.author.only", false);
106         deletableByAuthorOnly = config.getAsBoolean("comment.deletable.author.only", false);
107         add(new Label("doc-commenting-caption", getCaptionModel()));
108         MarkupContainer commentsContainer = new WebMarkupContainer("doc-comments-container");
109         addDialogAction = new DialogAction(
110                 createDialogFactory(new CommentItem(), (SerializableCallable<Object>) () -> {
111                     refreshCommentItems();
112                     return refreshDocumentEditorWithSelectedCompounds();
113                 }), getDialogService());
114 
115         AjaxLink<Void> addLink = new AjaxLink<Void>("add") {
116 
117             private static final long serialVersionUID = 1L;
118 
119             @Override
120             public void onClick(AjaxRequestTarget target) {
121                 addDialogAction.execute();
122             }
123         };
124 
125         final Image addImage = new Image("add-image") {
126             private static final long serialVersionUID = 1L;
127         };
128         addImage.setImageResourceReference(ADD_ICON_REF, null);
129         addLink.add(addImage);
130         addLink.setVisible(canCreateCommentItem(UserSession.get().getJcrSession()));
131         commentsContainer.add(addLink);
132         refreshCommentItems();
133         commentsContainer.add(createRefreshingView());
134         add(commentsContainer);
135     }
136 
137     private void refreshCommentItems() {
138         try {
139             String subjectId = getCommentingContext().getSubjectDocumentModel().getNode().getParent().getIdentifier();
140             List<CommentItem> commentItems = getCommentPersistenceManager().getLatestCommentItemsBySubjectId(getCommentingContext(),
141                     subjectId, 0, getQueryLimit());
142             currentCommentItems.clear();
143             currentCommentItems.addAll(commentItems);
144         } catch (RepositoryException e) {
145             log.error("Failed to refresh current comment items.", e);
146         }
147     }

163     private RefreshingView<? extends Serializable> createRefreshingView() {
164 
165         return new RefreshingView<>("view") {
166 
167             private static final long serialVersionUID = 1L;
168             private final IDataProvider<CommentItem> dataProvider = new SimpleListDataProvider<>(
169                     currentCommentItems);
170 
171             @Override
172             protected Iterator getItemModels() {
173                 final Iterator<? extends CommentItem> baseIt = dataProvider.iterator(0, currentCommentItems.size());
174                 return new Iterator<IModel<CommentItem>>() {
175                     public boolean hasNext() {
176                         return baseIt.hasNext();
177                     }
178 
179                     public IModel<CommentItem> next() {
180                         return dataProvider.model(baseIt.next());
181                     }
182 
183                     public void remove() {
184                         baseIt.remove();
185                     }
186                 };
187             }
188 
189             @Override
190             protected void populateItem(Item item) {
191                 final CommentItem comment = (CommentItem) item.getModelObject();
192                 final Session userJcrSession = UserSession.get().getJcrSession();
193 
194                 final Label commentHeadLabel = new Label("docitem-head-text", new Model<String>() {
195                     private static final long serialVersionUID = 1L;
196 
197                     @Override
198                     public String getObject() {
199                         return getCommentPersistenceManager().getCommentHeadText(getCommentingContext(), comment);
200                     }
201                 });
202                 commentHeadLabel.setEscapeModelStrings(false);
203                 item.add(commentHeadLabel);
204 
205                 final String headTooltip = getCommentPersistenceManager().getCommentHeadTooltip(getCommentingContext(), comment);
206                 if (StringUtils.isNotBlank(headTooltip)) {
207                     commentHeadLabel.add(new AttributeModifier("title", headTooltip));
208                 }
209 
210                 final Label commentBodyLabel = new Label("docitem-body-text", () ->
211                         getCommentPersistenceManager().getCommentBodyText(getCommentingContext(), comment));
212                 commentBodyLabel.setEscapeModelStrings(false);
213                 item.add(commentBodyLabel);
214 
215                 final String bodyTooltip = getCommentPersistenceManager().getCommentBodyTooltip(getCommentingContext(), comment);
216                 if (StringUtils.isNotBlank(bodyTooltip)) {
217                     commentBodyLabel.add(new AttributeModifier("title", bodyTooltip));
218                 }
219 
220                 if (item.getIndex() == currentCommentItems.size() - 1) {
221                     item.add(new AttributeAppender("class", "last", " "));
222                 }
223 
224                 final DialogAction editDialogAction = new DialogAction(
225                         createDialogFactory(comment, (SerializableCallable<Object>) () -> {
226                             refreshCommentItems();
227                             return refreshDocumentEditorWithSelectedCompounds();
228                         }), getDialogService());
229 
230                 AjaxLink<Void> editLink = new AjaxLink<Void>("edit") {
231 
232                     private static final long serialVersionUID = 1L;
233 
234                     @Override
235                     public void onClick(AjaxRequestTarget target) {
236                         editDialogAction.execute();
237                     }
238                 };
239 
240                 final Image editImage = new Image("edit-image") {
241                     private static final long serialVersionUID = 1L;
242                 };
243 
244                 editImage.setImageResourceReference(EDIT_ICON_REF, null);
245                 editLink.add(editImage);
246                 editLink.setVisible(isEditableCommentItem(userJcrSession, comment));
247                 item.add(editLink);
248 
249                 AjaxLink<Void> deleteLink = new AjaxLink<Void>("delete") {
250 
251                     private static final long serialVersionUID = 1L;
252 
253                     @Override
254                     public void onClick(final AjaxRequestTarget target) {
255                         try {
256                             final ConfirmDialog confirmDlg = new ConfirmDialog(
257                                     new StringResourceModel("confirm.delete.comment.title", this, null)
258                                             .setDefaultValue("Confirmation"),
259                                     new StringResourceModel("confirm.delete.comment.message", this, null)
260                                             .setDefaultValue("Are you sure to delete the item?")) {
261                                 @Override
262                                 public void invokeWorkflow() {
263                                     getCommentPersistenceManager().deleteCommentItem(getCommentingContext(), comment);
264                                     currentCommentItems.remove(comment);
265                                     refreshDocumentEditorWithSelectedCompounds();
266                                 }
267                             };
268                             getDialogService().show(confirmDlg);
269                         } catch (CommentingException e) {
270                             log.error("Failed to delete comment.", e);
271                         }
272                     }
273                 };
274 
275                 final Image deleteImage = new Image("delete-image") {
276                     private static final long serialVersionUID = 1L;
277                 };
278 
279                 deleteImage.setImageResourceReference(DELETE_ICON_REF, null);
280                 deleteLink.add(deleteImage);
281                 deleteLink.setVisible(isDeletableCommentItem(userJcrSession, comment));
282                 item.add(deleteLink);
283             }
284         };
285     }

289     }

292         return IEditor.Mode.COMPARE.equals(IEditor.Mode.fromString(getPluginConfig().getString("mode", "view")));
293     }

322     protected boolean canCreateCommentItem(final Session userJcrSession) {
323         return isEditMode();
324     }
325 
326     protected boolean isEditableCommentItem(final Session userJcrSession, final CommentItem commentItem) {
327         if (!isEditMode()) {
328             return false;
329         }
330         return !isEditableByAuthorOnly() || StringUtils.equals(commentItem.getAuthor(), userJcrSession.getUserID());
331     }
332 
333     protected boolean isDeletableCommentItem(final Session userJcrSession, final CommentItem commentItem) {
334         if (!isEditMode()) {
335             return false;
336         }
337         return !isDeletableByAuthorOnly() || StringUtils.equals(commentItem.getAuthor(), userJcrSession.getUserID());
338     }
339 
340     protected boolean isEditableByAuthorOnly() {
341         return editableByAuthorOnly;
342     }
343 
344     protected boolean isDeletableByAuthorOnly() {
345         return deletableByAuthorOnly;
346     }
347 
348     protected long getQueryLimit() {
349         return queryLimit;
350     }

352     private Object refreshDocumentEditorWithSelectedCompounds() {
353         // find the EditorForm and invoke #onModelChagned() in order to refresh the other editor form fields.
354         MarkupContainer container = this.getParent();
355         for (; container != null; container = container.getParent()) {
356             if (container instanceof EditorForm) {
357                 ((EditorForm) container).onModelChanged();
358                 break;
359             }
360         }
361         return null;
362     }
