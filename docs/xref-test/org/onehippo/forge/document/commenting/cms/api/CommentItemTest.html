25  public class CommentItemTest {
26  
27      @Test
28      public void testCloneAndEqualsAndHashCode() throws Exception {
29          CommentItem comment1 = new CommentItem();
30          comment1.setId("123");
31          comment1.setSubjectId("456");
32          comment1.setAuthor("author");
33          comment1.setCreated(Calendar.getInstance());
34          comment1.setLastModified(Calendar.getInstance());
35          comment1.setContent("Hello, World!");
36          comment1.setAttribute("attr1", "value1");
37          comment1.setAttribute("attr2", "value2");
38  
39          CommentItem comment2 = (CommentItem) comment1.clone();
40          assertEquals(comment1, comment2);
41          assertEquals(comment1.hashCode(), comment2.hashCode());
42          assertEquals(comment1.getId(), comment2.getId());
43          assertEquals(comment1.getSubjectId(), comment2.getSubjectId());
44          assertEquals(comment1.getAuthor(), comment2.getAuthor());
45          assertEquals(comment1.getCreated(), comment2.getCreated());
46          assertEquals(comment1.getLastModified(), comment2.getLastModified());
47          assertEquals(comment1.getContent(), comment2.getContent());
48          assertEquals(comment1.getAttribute("attr1"), comment1.getAttribute("attr1"));
49          assertEquals(comment1.getAttribute("attr2"), comment1.getAttribute("attr2"));
50  
51          comment2.setContent(comment2.getContent() + " " + comment2.getContent());
52          assertEquals("Hello, World! Hello, World!", comment2.getContent());
53          assertFalse(comment1.equals(comment2));
54          assertFalse(comment1.hashCode() == comment2.hashCode());
55      }
